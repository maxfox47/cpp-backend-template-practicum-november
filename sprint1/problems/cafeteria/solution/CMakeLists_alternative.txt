cmake_minimum_required(VERSION 3.16)

# Устанавливаем компиляторы
set(CMAKE_C_COMPILER gcc)
set(CMAKE_CXX_COMPILER g++)

project(cafeteria_solution)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Найти Boost
find_package(Boost REQUIRED COMPONENTS system)

# Найти pthread
find_package(Threads REQUIRED)

# Создать исполняемый файл
add_executable(cafeteria src/main.cpp)

# Подключить заголовочные файлы
target_include_directories(cafeteria PRIVATE src)

# Подключить библиотеки
target_link_libraries(cafeteria 
    Boost::system 
    Threads::Threads
)

# Настройки компилятора
target_compile_options(cafeteria PRIVATE -Wall -Wextra -Wpedantic)
